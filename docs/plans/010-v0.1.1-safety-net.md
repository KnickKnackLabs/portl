```rust
use nix::pty::{openpty, OpenptyResult};
use nix::unistd::{setsid, dup2, close};
use std::os::unix::io::{AsRawFd, FromRawFd, OwnedFd};
use std::os::unix::process::CommandExt;
use tokio::process::{Child, Command as TokioCommand};

pub(crate) async fn spawn_pty(
    program: &str,
    argv: &[&str],
    size: PtySize,
    env: Vec<(String, String)>,
) -> std::io::Result<(OwnedFd, Child)> {
    let OpenptyResult { master, slave } = openpty(Some(&size.into()), None)
        .map_err(std::io::Error::from)?;
    let slave_fd = slave.as_raw_fd();

    let mut cmd = TokioCommand::new(program);
    cmd.args(argv);
    for (k, v) in env {
        cmd.env(k, v);
    }
    unsafe {
        cmd.pre_exec(move || {
            setsid().map_err(std::io::Error::from)?;
            // make slave the controlling terminal
            unsafe {
                libc::ioctl(slave_fd, libc::TIOCSCTTY as _, 0);
            }
            dup2(slave_fd, 0)?;
            dup2(slave_fd, 1)?;
            dup2(slave_fd, 2)?;
            if slave_fd > 2 {
                let _ = close(slave_fd);
            }
            Ok(())
        });
    }
    let child = cmd.spawn()?;
    drop(slave); // close slave in the parent
    Ok((master, child))
}

#[cfg(test)]
pub(crate) use spawn_pty as spawn_pty_for_test;
```

```rust
#[cfg(unix)]
pub(crate) fn apply_rlimits() -> std::io::Result<()> {
    use libc::{RLIMIT_NOFILE, RLIMIT_CORE, RLIMIT_CPU, RLIMIT_FSIZE, rlimit, setrlimit};
    unsafe {
        let set = |res: libc::__rlimit_resource_t, v: u64| {
            let rl = rlimit { rlim_cur: v, rlim_max: v };
            if setrlimit(res, &rl) != 0 {
                return Err(std::io::Error::last_os_error());
            }
            Ok(())
        };
        set(RLIMIT_NOFILE, 4096)?;
        set(RLIMIT_CORE,   0)?;
        set(RLIMIT_CPU,    86_400)?;
        set(RLIMIT_FSIZE,  10 * 1024 * 1024 * 1024)?;
        #[cfg(target_os = "linux")]
        {
            set(libc::RLIMIT_NPROC, 512)?;
        }
    }
    Ok(())
}
```

Call `apply_rlimits()?;` as the first statement of both `pre_exec`
closures (exec path + PTY path from Task 2). Linux-only `NPROC` is
